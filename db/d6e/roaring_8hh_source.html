868 static Roaring fastunion(size_t n, const Roaring **inputs) {
869 const roaring_bitmap_t **x = (const roaring_bitmap_t **)roaring_malloc(
870 n * sizeof(roaring_bitmap_t *));
871 if (x == NULL) {
872 ROARING_TERMINATE("failed memory alloc in fastunion");
873 }
874 for (size_t k = 0; k < n; ++k) x[k] = &inputs[k]->roaring;
875
876 roaring_bitmap_t *c_ans = api::roaring_bitmap_or_many(n, x);
877 if (c_ans == NULL) {
878 roaring_free(x);
879 ROARING_TERMINATE("failed memory alloc in fastunion");
880 }
881 Roaring ans(c_ans);
882 roaring_free(x);
883 return ans;
884 }
