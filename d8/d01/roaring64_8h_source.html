70// Use an immediately invoked closure, capturing by reference
71// (in case __VA_ARGS__ refers to context outside the closure)
72// Include a 0 at the beginning of the array to make the array length > 0
73// (zero sized arrays are not valid in standard c/c++)
74#define roaring64_bitmap_from(...) \
75 [&]() { \
76 const uint64_t roaring64_bitmap_from_array[] = {0, __VA_ARGS__}; \
77 return roaring64_bitmap_of_ptr( \
78 (sizeof(roaring64_bitmap_from_array) / \
79 sizeof(roaring64_bitmap_from_array[0])) - \
80 1, \

82 }()
